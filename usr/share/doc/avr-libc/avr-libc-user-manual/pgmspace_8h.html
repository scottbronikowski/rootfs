#define pgm_get_far_address (   var)
Value:
({                                                    \
        uint_farptr_t tmp;                                \
                                                      \
        __asm__ __volatile__(                             \
                                                      \
                        "ldi    %A0, lo8(%1)"           "\n\t"    \
                        "ldi    %B0, hi8(%1)"           "\n\t"    \
                        "ldi    %C0, hh8(%1)"           "\n\t"    \
                        "clr    %D0"                    "\n\t"    \
                :                                             \
                        "=d" (tmp)                                \
                :                                             \
                        "p"  (&(var))                             \
        );                                                \
        tmp;                                              \
})
